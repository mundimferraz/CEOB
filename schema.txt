
-- SCRIPT DE ATUALIZAÇÃO SGR-VIAS --
-- Execute este script no SQL Editor do Supabase para sincronizar a estrutura --

-- 1. Tabela de Usuários (Se não existir)
CREATE TABLE IF NOT EXISTS users (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    role TEXT NOT NULL,
    zonal TEXT NOT NULL,
    registration_number TEXT,
    email TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 2. Regra de Unicidade: Apenas 1 Engenheiro (Manager) por Zonal
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_class WHERE relname = 'unique_manager_per_zonal') THEN
        CREATE UNIQUE INDEX unique_manager_per_zonal ON users (zonal) WHERE (role = 'Manager');
    END IF;
END $$;

-- 3. Tabela de Zonais (Se não existir)
CREATE TABLE IF NOT EXISTS zonals (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    manager_id TEXT REFERENCES users(id) ON DELETE SET NULL,
    assistant_id TEXT REFERENCES users(id) ON DELETE SET NULL,
    description TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Adicionar coluna assistant_id caso a tabela já exista
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='zonals' AND column_name='assistant_id') THEN
        ALTER TABLE zonals ADD COLUMN assistant_id TEXT REFERENCES users(id) ON DELETE SET NULL;
    END IF;
END $$;

-- 4. Tabela de Vistorias (Se não existir)
CREATE TABLE IF NOT EXISTS repair_requests (
    id TEXT PRIMARY KEY,
    protocol TEXT NOT NULL,
    sei_number TEXT NOT NULL,
    contract TEXT NOT NULL,
    description TEXT,
    latitude DOUBLE PRECISION NOT NULL,
    longitude DOUBLE PRECISION NOT NULL,
    address TEXT,
    visit_date DATE NOT NULL,
    status TEXT NOT NULL,
    technician_id TEXT REFERENCES users(id),
    zonal TEXT NOT NULL,
    photo_before TEXT,
    photo_after TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 5. Adição de colunas extras (caso a tabela já existisse mas estivesse incompleta)
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM information_schema.columns WHERE table_name='repair_requests' AND column_name='photo_after') THEN
        ALTER TABLE repair_requests ADD COLUMN photo_after TEXT;
    END IF;
END $$;
